    **handleSave:**
    ```typescript
    function handleSave() {
      const now = new Date().toISOString();
      if (mode === "create") {
        addPin({
          id: crypto.randomUUID(),
          title: title.trim() || address || "New Pin",
          address, status, contact, phone, followUpDate, notes,
          lat: initialData.lat ?? 0,
          lng: initialData.lng ?? 0,
          createdAt: now, updatedAt: now,
        });
      } else {
        updatePin(initialData.id!, {
          title, address, status, contact, phone, followUpDate, notes,
          updatedAt: now,
        });
      }
      onClose();
    }
    ```

    **handleDelete** (edit mode only):
    ```typescript
    function handleDelete() {
      if (initialData.id) { deletePin(initialData.id); }
      onClose();
    }
    ```

    **handleAddNote:**
    ```typescript
    function handleAddNote() {
      const text = noteInput.trim();
      if (!text) return;
      setNotes((prev) => [...prev, { text, date: new Date().toISOString() }]);
      setNoteInput("");
    }
    ```
