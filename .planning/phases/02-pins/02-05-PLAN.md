    **PinListItem.tsx:**

    Create `app/features/pins/PinListItem.tsx` as a "use client" component.

    Props: `interface PinListItemProps { pin: Pin; markerPoolRef: React.RefObject<Map<string, google.maps.marker.AdvancedMarkerElement>>; }`

    The `markerPoolRef` is needed to access the marker DOM element for the bounce animation.
    However, markerPoolRef lives inside MarkerLayer which is a sibling component. Passing it from a shared parent is complex at this stage. Instead, use this simpler approach:

    Bounce via CSS class on the marker element accessed through the DOM:
    ```typescript
    function handleClick() {
      const map = useMapInstance(); // ← call inside component body, not here
      ...
    }
    ```

    Actually structure as follows — use the map from hook and locate the marker element by data attribute:

    ```typescript
    "use client";
    import { useMapInstance } from "@/app/features/map/MapContext";
    import type { Pin } from "@/app/types/pins.types";

    interface PinListItemProps {
      pin: Pin;
      onEditPin: (pinId: string) => void;
    }

    export function PinListItem({ pin, onEditPin }: PinListItemProps) {
      const map = useMapInstance();

      function handleClick() {
        if (!map) return;
        map.panTo({ lat: pin.lat, lng: pin.lng });
        map.setZoom(15);
        // Find marker element by data-pin-id attribute and bounce it
        const markerEl = document.querySelector<HTMLElement>(`[data-pin-id="${pin.id}"]`);
        if (markerEl) {
          markerEl.classList.add("marker-bounce");
          setTimeout(() => markerEl.classList.remove("marker-bounce"), 700);
        }
      }

      const STATUS_COLORS: Record<string, string> = {
        prospect: "#3B82F6", active: "#22C55E", "follow-up": "#F59E0B", lost: "#EF4444",
      };
      const STATUS_LABELS: Record<string, string> = {
        prospect: "Prospect", active: "Active", "follow-up": "Follow-Up", lost: "Lost",
      };

      return (
        <div
          className="flex items-center gap-3 px-3 py-2.5 rounded-lg cursor-pointer transition-colors duration-150 hover:bg-orange-dim group"
          onClick={handleClick}
        >
          <span
            className="w-2.5 h-2.5 rounded-full shrink-0"
            style={{ background: STATUS_COLORS[pin.status] }}
          />
          <div className="flex-1 min-w-0">
            <div className="text-sm font-semibold text-text-primary truncate">{pin.title || "Unnamed Pin"}</div>
            <div className="text-[11px] text-text-secondary truncate mt-0.5">{pin.address}</div>
          </div>
          <button
            className="opacity-0 group-hover:opacity-100 shrink-0 text-text-muted hover:text-orange transition-opacity"
            onClick={(e) => { e.stopPropagation(); onEditPin(pin.id); }}
            title="Edit pin"
          >
            <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
              <path d="M11 4H4a2 2 0 00-2 2v14a2 2 0 002 2h14a2 2 0 002-2v-7"/>
              <path d="M18.5 2.5a2.121 2.121 0 013 3L12 15l-4 1 1-4 9.5-9.5z"/>
            </svg>
          </button>
        </div>
      );
    }
    ```

    Filtered list (useMemo):
    ```typescript
    const filtered = useMemo(() => {
      const q = searchText.toLowerCase().trim();
      return pins.filter((p) => {
        const matchesSearch =
          !q ||
          p.title.toLowerCase().includes(q) ||
          p.address.toLowerCase().includes(q) ||
          p.contact.toLowerCase().includes(q);
        const matchesStatus = activeStatusFilter.has(p.status);
        return matchesSearch && matchesStatus;
      });
    }, [pins, searchText, activeStatusFilter]);
    ```

    toggleStatus handler:
    ```typescript
    function toggleStatus(status: PinStatus) {
      const next = new Set(activeStatusFilter);
      if (next.has(status)) {
        next.delete(status);
      } else {
        next.add(status);
      }
      setActiveStatusFilter(next);
    }
    ```

    Stats computation (useMemo):
    - totalPins: pins.length
    - activeCount: pins.filter(p => p.status === "active").length
    - thisWeekCount: pins with followUpDate in the next 7 days
    - overdueCount: pins with followUpDate before today and not empty

    JSX structure for PinList:
    ```
    <div className="flex flex-col h-full">
      {/* Search input */}
      <div className="px-4 py-3 border-b border-border bg-bg-card relative">
        [search input — matches the static HTML in Sidebar.tsx, but now controlled]
        <input value={searchText} onChange={(e) => setSearchText(e.target.value)} ... />
      </div>

      {/* Pin count row */}
      <div className="flex items-center justify-between px-4 py-2 text-xs text-text-muted border-b border-border font-medium">
        <span>{filtered.length} pin{filtered.length !== 1 ? "s" : ""}</span>
      </div>

      {/* Status filter chips */}
      <div className="px-3 py-2 border-b border-border bg-bg-card">
        <div className="flex flex-wrap gap-[5px]">
          {[
            { status: "prospect" as PinStatus, label: "Prospect", color: "#3B82F6" },
            { status: "active" as PinStatus, label: "Active", color: "#22C55E" },
            { status: "follow-up" as PinStatus, label: "Follow-Up", color: "#F59E0B" },
            { status: "lost" as PinStatus, label: "Lost", color: "#EF4444" },
          ].map(({ status, label, color }) => {
            const count = pins.filter((p) => p.status === status).length;
            const isActive = activeStatusFilter.has(status);
            return (
              <button
                key={status}
                onClick={() => toggleStatus(status)}
                className={`flex items-center gap-1 px-2.5 py-1 rounded-full text-[11px] font-semibold cursor-pointer transition-all duration-150 border-[1.5px] ${
                  isActive
                    ? "border-orange bg-orange-dim text-orange"
                    : "border-border text-text-secondary bg-bg-input hover:border-text-muted"
                }`}
              >
                <span className="w-[7px] h-[7px] rounded-full" style={{ background: color }} />
                {label} <span className="font-normal opacity-70">{count}</span>
              </button>
            );
          })}
        </div>
      </div>

      {/* Pin list */}
      <div className="flex-1 overflow-y-auto p-2 scrollbar-thin">
        {filtered.length === 0 ? (
          <EmptyState searchText={searchText} />
        ) : (
          filtered.map((pin) => (
            <PinListItem key={pin.id} pin={pin} onEditPin={onEditPin} />
          ))
        )}
      </div>

      {/* Stats footer */}
      <div className="flex shrink-0 px-4 py-2.5 border-t border-border bg-bg-card">
        {[
          { num: String(totalPins), label: "Pins" },
          { num: String(activeCount), label: "Active" },
          { num: String(thisWeekCount), label: "This Week" },
          { num: String(overdueCount), label: "Overdue" },
        ].map((stat, i, arr) => (
          <div key={stat.label} className={`flex-1 text-center py-1.5 px-1 rounded-md transition-colors duration-150 hover:bg-orange-dim ${i < arr.length - 1 ? "border-r border-border" : ""}`}>
            <div className="text-lg font-extrabold text-orange leading-tight tabular-nums">{stat.num}</div>
            <div className="text-[9px] font-bold uppercase tracking-wide text-text-muted mt-0.5">{stat.label}</div>
          </div>
        ))}
      </div>
    </div>
    ```

    EmptyState: a small inline component (same file) showing the existing "No pins yet" SVG message when filtered.length === 0 and searchText is empty, or "No pins match your search" when searchText is non-empty.
