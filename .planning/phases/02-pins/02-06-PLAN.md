  <action>
    Rewrite `app/page.tsx` to become a "use client" component that coordinates the edit modal state:

    ```typescript
    "use client";
    import { useState, useCallback } from "react";
    import Sidebar from "./components/Sidebar";
    import Map from "./features/map/Map";
    import MobileBottomBar from "./components/MobileBottomBar";
    import StoreHydration from "./components/StoreHydration";
    import PinModal from "./features/pins/PinModal";
    import { useStore } from "@/app/store";

    export default function Home() {
      const [editPinId, setEditPinId] = useState<string | null>(null);
      const pins = useStore((s) => s.pins);

      const openEditModal = useCallback((pinId: string) => {
        setEditPinId(pinId);
      }, []);

      const editPin = editPinId ? (pins.find((p) => p.id === editPinId) ?? null) : null;

      return (
        <>
          <StoreHydration />
          <div className="flex h-screen w-screen">
            <Sidebar onEditPin={openEditModal} />
            <Map onEditPin={openEditModal} />
          </div>
          <MobileBottomBar />
          {editPin && (
            <PinModal
              mode="edit"
              initialData={editPin}
              onClose={() => setEditPinId(null)}
            />
          )}
        </>
      );
    }
    ```

    Key points:
    - `StoreHydration` is rendered FIRST (before Sidebar and Map) so pins load from localStorage before the list renders
    - `openEditModal` is passed to both Sidebar (for sidebar's edit button on PinListItem) and Map (for MarkerLayer's info window Edit button)
    - The edit-mode PinModal is rendered at the page level (outside Map) — clean z-index, single source of truth
    - The create-mode PinModal is rendered inside Map.tsx (because it needs pendingPin state which is internal to Map)
    - Both modals use createPortal into document.body so they escape any parent stacking context
